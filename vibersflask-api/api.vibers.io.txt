server {

        root /home/restrictedUser/vibersflask-api/html;
        index index.html index.htm index.nginx-debian.html;

        server_name api.vibers.io www.api.vibers.io;

        location / {
            proxy_pass http://localhost:3000;
            proxy_http_version 1.1;
            proxy_set_header Upgrade $http_upgrade;
            proxy_set_header Connection 'upgrade';
            proxy_set_header Host $host;
            proxy_cache_bypass $http_upgrade;
        }

        listen [::]:443 ssl; # managed by Certbot
        listen 443 ssl; # managed by Certbot
        ssl_certificate /etc/letsencrypt/live/api.vibers.io/fullchain.pem; # managed by Certbot
        ssl_certificate_key /etc/letsencrypt/live/api.vibers.io/privkey.pem; # managed by Certbot
        include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
        ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot

}
server {
    if ($host = www.api.vibers.io) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


    if ($host = api.vibers.io) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


        listen 80;
        listen [::]:80;

        server_name api.vibers.io www.api.vibers.io;
    return 404; # managed by Certbot

}
